## 1. 需求总览与分析

### 1.1 核心功能矩阵

| 功能域  | 功能点                      | 优先级 |
|------|--------------------------|-----|
| 拖拽面板 | 左侧 Palette 展示所有可用 Token  | P0  |
| 拖拽面板 | 自定义常量/变量输入               | P0  |
| 拖拽面板 | 子表达式库                    | P1  |
| 公式画布 | 拖入 Token、排序、删除、复制        | P0  |
| 公式画布 | 结构化模板自动展开（如果→那么→否则）      | P0  |
| 公式画布 | 插入时自动让位动画                | P0  |
| 公式画布 | 实时校验与报红                  | P0  |
| 公式画布 | 可编辑 Token（自定义常量/变量可直接改值） | P0  |
| 编译输出 | 前端展示公式（中文可读）             | P0  |
| 编译输出 | 后端标准表达式（JS 三元/逻辑运算符）     | P0  |
| 子表达式 | 框选/全选保存为命名子表达式           | P1  |
| 子表达式 | 子表达式拖入画布时展开 or 折叠显示      | P1  |

### 1.2 Token 分类总表

```
┌─────────────────────────────────────────────────┐
│                 Token 分类                        │
├──────────┬──────────────────────────────────────┤
│ 控制流    │ 如果(IF)、那么(THEN)、否则(ELSE)       │
├──────────┼──────────────────────────────────────┤
│ 逻辑运算符│ 并且(AND)、或者(OR)、非(NOT)           │
├──────────┼──────────────────────────────────────┤
│ 比较运算符│ ==(EQ)、!=(NEQ)、>(GT)、<(LT)、       │
│          │ >=(GTE)、<=(LTE)                      │
├──────────┼──────────────────────────────────────┤
│ 算术运算符│ +(ADD)、-(SUB)、*(MUL)、/(DIV)、      │
│          │ %(MOD)                                │
├──────────┼──────────────────────────────────────┤
│ 分组      │ 左括号(LPAREN)、右括号(RPAREN)         │
├──────────┼──────────────────────────────────────┤
│ 值类型    │ 常量(CONSTANT)、变量(VARIABLE)、       │
│          │ 子表达式(SUB_EXPR)                     │
├──────────┼──────────────────────────────────────┤
│ 占位符    │ 空位(PLACEHOLDER) — 仅画布内部使用     │
└──────────┴──────────────────────────────────────┘
```

### 1.3 输入→输出映射示例

**画布可视：**

```
【如果】 【a】【==】【100】 【并且】 【(】【条件b】【或者】【条件c】【)】 【那么】 【值1】【*】【100】【+】【1】 【否则】 【值2】【/】【2】【+】【300】
```

**后端表达式：**

```
a==100&&(条件b||条件c)?值1*100+1:值2/2+300
```

### 2.3 技术选型细化

| 能力    | 方案                                    | 理由                         |
|-------|---------------------------------------|----------------------------|
| 拖拽核心  | 原生 HTML5 Drag & Drop + 自定义 composable | 避免引入重型库；需精细控制插入位置与动画       |
| 动画    | Vue `<TransitionGroup>` + FLIP 动画     | 列表重排动画流畅                   |
| 状态管理  | Pinia                                 | 项目已有；Token 列表、校验状态、历史栈统一管理 |
| 校验    | 自研有限状态机 (FSM)                         | 规则复杂且需实时；无现成库              |
| 编译    | 自研递归下降 / Token→AST→CodeGen            | 确保输出合法表达式                  |
| 图标    | lucide-vue-next                       | 项目已有                       |
| UI 组件 | Element Plus                          | 项目已有；对话框、输入框、提示等复用         |

## 3. 数据模型设计（核心 Token 体系）

### 3.1 Token 基础类型定义

```typescript
// ==================== types/formula.ts ====================

/** Token 大类 */
export enum TokenCategory {
  CONTROL_FLOW = 'CONTROL_FLOW',     // 控制流：IF/THEN/ELSE
  LOGIC_OP     = 'LOGIC_OP',         // 逻辑运算符
  COMPARE_OP   = 'COMPARE_OP',       // 比较运算符
  ARITHMETIC_OP= 'ARITHMETIC_OP',    // 算术运算符
  GROUPING     = 'GROUPING',         // 括号
  VALUE        = 'VALUE',            // 值：常量/变量
  PLACEHOLDER  = 'PLACEHOLDER',      // 占位符
  SUB_EXPR     = 'SUB_EXPR',         // 子表达式（折叠块）
}

/** Token 具体类型 */
export enum TokenType {
  // 控制流
  IF       = 'IF',
  THEN     = 'THEN',
  ELSE     = 'ELSE',

  // 逻辑
  AND      = 'AND',
  OR       = 'OR',
  NOT      = 'NOT',

  // 比较
  EQ       = 'EQ',       // ==
  NEQ      = 'NEQ',      // !=
  GT       = 'GT',       // >
  LT       = 'LT',       // <
  GTE      = 'GTE',      // >=
  LTE      = 'LTE',      // <=

  // 算术
  ADD      = 'ADD',       // +
  SUB      = 'SUB',       // -
  MUL      = 'MUL',       // *
  DIV      = 'DIV',       // /
  MOD      = 'MOD',       // %

  // 括号
  LPAREN   = 'LPAREN',    // (
  RPAREN   = 'RPAREN',    // )

  // 值
  CONSTANT = 'CONSTANT',  // 常量（数字/字符串）
  VARIABLE = 'VARIABLE',  // 变量

  // 占位符
  PLACEHOLDER = 'PLACEHOLDER',

  // 子表达式
  SUB_EXPR = 'SUB_EXPR',
}

/** 单个 Token 实例 */
export interface FormulaToken {
  /** 唯一标识（画布内实例级别） */
  id: string

  /** Token 类型 */
  type: TokenType

  /** Token 大类（冗余，方便判断） */
  category: TokenCategory

  /** 显示文本（中文可读） */
  displayText: string

  /** 输出值（编译为后端表达式时的值） */
  outputValue: string

  /** 是否可编辑（自定义常量/变量可编辑） */
  editable: boolean

  /** 是否是结构化模板的一部分 */
  templateGroupId?: string

  /** 校验状态 */
  validationState: 'valid' | 'error' | 'warning'

  /** 校验错误信息 */
  validationMessage?: string

  /** 子表达式 ID（当 type === SUB_EXPR 时） */
  subExprId?: string

  /** 子表达式展开后的 Token 列表（运行时填充） */
  subExprTokens?: FormulaToken[]

  /** 嵌套深度（用于缩进/颜色区分） */
  nestingLevel: number

  /** 创建时间（用于排序稳定性） */
  createdAt: number
}

/** 模板组 - 结构化 Token 组 */
export interface TemplateGroup {
  /** 组 ID */
  id: string

  /** 模板类型 */
  templateType: 'IF_THEN_ELSE' // 将来可扩展更多模板

  /** 组内 Token ID 列表（有序） */
  tokenIds: string[]

  /** 占位符 Token ID 映射 */
  slots: {
    condition: string    // IF 后面的占位符
    trueBranch: string   // THEN 后面的占位符
    falseBranch: string  // ELSE 后面的占位符
  }
}

/** 子表达式定义 */
export interface SubExpression {
  /** 唯一 ID */
  id: string

  /** 用户自定义名称 */
  name: string

  /** 描述 */
  description?: string

  /** 包含的 Token 列表（原始数据，深拷贝） */
  tokens: FormulaToken[]

  /** 编译后的后端表达式（缓存） */
  compiledExpression: string

  /** 显示用的中文表达式（缓存） */
  displayExpression: string

  /** 创建时间 */
  createdAt: number

  /** 更新时间 */
  updatedAt: number
}

/** 拖拽数据载荷 */
export interface DragPayload {
  /** 来源 */
  source: 'palette' | 'canvas'

  /** 来自 palette 时的 Token 原型 */
  tokenPrototype?: Omit<FormulaToken, 'id' | 'createdAt' | 'validationState' | 'nestingLevel'>

  /** 来自 canvas 时的 Token ID */
  tokenId?: string

  /** 是否是模板类型（IF 拖入需要自动展开） */
  isTemplate?: boolean

  /** 模板类型 */
  templateType?: 'IF_THEN_ELSE'
}

/** Palette 项定义 */
export interface PaletteItem {
  /** 显示文本 */
  label: string

  /** Token 类型 */
  tokenType: TokenType

  /** Token 大类 */
  category: TokenCategory

  /** 编译输出值 */
  outputValue: string

  /** 图标名称 */
  icon?: string

  /** 是否是模板 */
  isTemplate?: boolean

  /** 模板类型 */
  templateType?: 'IF_THEN_ELSE'

  /** 是否可编辑 */
  editable?: boolean
}
```

### 3.3 Pinia Store 核心结构

```typescript
// ==================== stores/formulaStore.ts ====================

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { FormulaToken, TemplateGroup, SubExpression } from '@/types/formula'

/** 历史记录快照 */
interface HistorySnapshot {
  tokens: FormulaToken[]
  templateGroups: TemplateGroup[]
  timestamp: number
}

export const useFormulaStore = defineStore('formula', () => {
  // ═══════════════ State ═══════════════

  /** 画布上的 Token 列表（有序） */
  const tokens = ref<FormulaToken[]>([])

  /** 模板组注册表 */
  const templateGroups = ref<Map<string, TemplateGroup>>(new Map())

  /** 预定义变量列表（来自后端/配置） */
  const predefinedVariables = ref<Array<{ name: string; label: string; description?: string }>>([])

  /** 用户自定义常量/变量列表 */
  const customValues = ref<Array<{
    id: string
    type: 'CONSTANT' | 'VARIABLE'
    name: string
    value: string
  }>>([])

  /** 子表达式库 */
  const subExpressions = ref<SubExpression[]>([])

  /** 历史栈（撤销） */
  const undoStack = ref<HistorySnapshot[]>([])

  /** 重做栈 */
  const redoStack = ref<HistorySnapshot[]>([])

  /** 最大历史记录数 */
  const MAX_HISTORY = 50

  /** 当前拖拽状态 */
  const dragState = ref<{
    isDragging: boolean
    payload: any | null
    dropTargetIndex: number | null  // 当前悬停的插入位置
    dropTargetTokenId: string | null
  }>({
    isDragging: false,
    payload: null,
    dropTargetIndex: null,
    dropTargetTokenId: null,
  })

  // ═══════════════ Getters ═══════════════

  /** 可见 Token（不含被折叠的子表达式内部 Token） */
  const visibleTokens = computed(() => {
    return tokens.value
  })

  /** 是否有校验错误 */
  const hasErrors = computed(() => {
    return tokens.value.some(t => t.validationState === 'error')
  })

  /** 编译后的中文可读表达式 */
  const displayExpression = computed(() => {
    return tokens.value
      .filter(t => t.type !== 'PLACEHOLDER')
      .map(t => t.displayText)
      .join(' ')
  })

  /** 编译后的后端表达式 */
  const compiledExpression = computed(() => {
    // 委托给编译器模块（在第6章详述）
    return compileTokensToExpression(tokens.value)
  })

  // ═══════════════ Actions ═══════════════

  /** 保存快照到历史栈 */
  function saveSnapshot() {
    const snapshot: HistorySnapshot = {
      tokens: JSON.parse(JSON.stringify(tokens.value)),
      templateGroups: JSON.parse(JSON.stringify(Array.from(templateGroups.value.entries()))),
      timestamp: Date.now(),
    }
    undoStack.value.push(snapshot)
    if (undoStack.value.length > MAX_HISTORY) {
      undoStack.value.shift()
    }
    redoStack.value = [] // 新操作清空重做栈
  }

  /** 撤销 */
  function undo() {
    if (undoStack.value.length === 0) return
    const current: HistorySnapshot = {
      tokens: JSON.parse(JSON.stringify(tokens.value)),
      templateGroups: JSON.parse(JSON.stringify(Array.from(templateGroups.value.entries()))),
      timestamp: Date.now(),
    }
    redoStack.value.push(current)
    const prev = undoStack.value.pop()!
    tokens.value = prev.tokens
    templateGroups.value = new Map(prev.templateGroups)
  }

  /** 重做 */
  function redo() {
    if (redoStack.value.length === 0) return
    const current: HistorySnapshot = {
      tokens: JSON.parse(JSON.stringify(tokens.value)),
      templateGroups: JSON.parse(JSON.stringify(Array.from(templateGroups.value.entries()))),
      timestamp: Date.now(),
    }
    undoStack.value.push(current)
    const next = redoStack.value.pop()!
    tokens.value = next.tokens
    templateGroups.value = new Map(next.templateGroups)
  }

  /** 在指定位置插入 Token */
  function insertToken(token: FormulaToken, index: number) {
    saveSnapshot()
    tokens.value.splice(index, 0, token)
    runValidation()
  }

  /** 在指定位置插入多个 Token（模板展开） */
  function insertTokens(newTokens: FormulaToken[], index: number) {
    saveSnapshot()
    tokens.value.splice(index, 0, ...newTokens)
    runValidation()
  }

  /** 删除 Token */
  function removeToken(tokenId: string) {
    saveSnapshot()
    const token = tokens.value.find(t => t.id === tokenId)
    if (!token) return

    // 如果属于模板组，整组删除
    if (token.templateGroupId) {
      const group = templateGroups.value.get(token.templateGroupId)
      if (group) {
        tokens.value = tokens.value.filter(t => t.templateGroupId !== token.templateGroupId)
        templateGroups.value.delete(token.templateGroupId)
      }
    } else {
      tokens.value = tokens.value.filter(t => t.id !== tokenId)
    }
    runValidation()
  }

  /** 移动 Token（画布内排序） */
  function moveToken(fromIndex: number, toIndex: number) {
    saveSnapshot()
    const [moved] = tokens.value.splice(fromIndex, 1)
    // 如果是模板组成员，需整组移动
    if (moved.templateGroupId) {
      // 整组移动逻辑（稍后详述）
      handleTemplateMoveGroup(moved, fromIndex, toIndex)
    } else {
      const adjustedIndex = toIndex > fromIndex ? toIndex - 1 : toIndex
      tokens.value.splice(adjustedIndex, 0, moved)
    }
    runValidation()
  }

  /** 复制 Token */
  function duplicateToken(tokenId: string) {
    saveSnapshot()
    const index = tokens.value.findIndex(t => t.id === tokenId)
    if (index === -1) return
    const original = tokens.value[index]

    if (original.templateGroupId) {
      // 复制整个模板组
      duplicateTemplateGroup(original.templateGroupId, index)
    } else {
      const cloned = cloneToken(original)
      tokens.value.splice(index + 1, 0, cloned)
    }
    runValidation()
  }

  /** 更新 Token 值（用于可编辑 Token） */
  function updateTokenValue(tokenId: string, newDisplayText: string, newOutputValue: string) {
    const token = tokens.value.find(t => t.id === tokenId)
    if (token && token.editable) {
      saveSnapshot()
      token.displayText = newDisplayText
      token.outputValue = newOutputValue
      runValidation()
    }
  }

  /** 清空画布 */
  function clearCanvas() {
    saveSnapshot()
    tokens.value = []
    templateGroups.value.clear()
  }

  /** 替换占位符 */
  function replacePlaceholder(placeholderId: string, newTokens: FormulaToken[]) {
    const index = tokens.value.findIndex(t => t.id === placeholderId)
    if (index === -1) return
    saveSnapshot()
    tokens.value.splice(index, 1, ...newTokens)
    runValidation()
  }

  /** 触发校验 */
  function runValidation() {
    // 委托给校验引擎（第5章详述）
    validateTokenSequence(tokens.value)
  }

  return {
    // state
    tokens,
    templateGroups,
    predefinedVariables,
    customValues,
    subExpressions,
    dragState,
    undoStack,
    redoStack,
    // getters
    visibleTokens,
    hasErrors,
    displayExpression,
    compiledExpression,
    // actions
    insertToken,
    insertTokens,
    removeToken,
    moveToken,
    duplicateToken,
    updateTokenValue,
    clearCanvas,
    replacePlaceholder,
    runValidation,
    undo,
    redo,
    saveSnapshot,
  }
})
```

```typescript
// ==================== utils/tokenFactory.ts ====================

import { v4 as uuidv4 } from 'uuid' // 或使用 crypto.randomUUID()
import { TokenType, TokenCategory, type FormulaToken, type TemplateGroup } from '@/types/formula'
import { TOKEN_META } from '@/config/tokenRegistry'

/**
 * 创建单个 Token 实例
 */
export function createToken(
  type: TokenType,
  overrides?: Partial<FormulaToken>
): FormulaToken {
  const meta = TOKEN_META[type]
  return {
    id: uuidv4(),
    type,
    category: meta.category,
    displayText: meta.displayText,
    outputValue: meta.outputValue,
    editable: meta.editable,
    validationState: 'valid',
    nestingLevel: 0,
    createdAt: Date.now(),
    ...overrides,
  }
}

/**
 * 创建自定义常量 Token
 */
export function createConstantToken(value: string): FormulaToken {
  return createToken(TokenType.CONSTANT, {
    displayText: value,
    outputValue: value,
    editable: true,
  })
}

/**
 * 创建自定义变量 Token
 */
export function createVariableToken(name: string, outputName?: string): FormulaToken {
  return createToken(TokenType.VARIABLE, {
    displayText: name,
    outputValue: outputName || name,
    editable: true,
  })
}

/**
 * 创建 IF-THEN-ELSE 模板组
 * 返回一组 Token 和 TemplateGroup 定义
 */
export function createIfThenElseTemplate(): {
  tokens: FormulaToken[]
  group: TemplateGroup
} {
  const groupId = uuidv4()

  const ifToken = createToken(TokenType.IF, { templateGroupId: groupId })
  const condPlaceholder = createToken(TokenType.PLACEHOLDER, {
    templateGroupId: groupId,
    displayText: '条件',
  })
  const thenToken = createToken(TokenType.THEN, { templateGroupId: groupId })
  const truePlaceholder = createToken(TokenType.PLACEHOLDER, {
    templateGroupId: groupId,
    displayText: '真值',
  })
  const elseToken = createToken(TokenType.ELSE, { templateGroupId: groupId })
  const falsePlaceholder = createToken(TokenType.PLACEHOLDER, {
    templateGroupId: groupId,
    displayText: '假值',
  })

  const tokens = [ifToken, condPlaceholder, thenToken, truePlaceholder, elseToken, falsePlaceholder]

  const group: TemplateGroup = {
    id: groupId,
    templateType: 'IF_THEN_ELSE',
    tokenIds: tokens.map(t => t.id),
    slots: {
      condition: condPlaceholder.id,
      trueBranch: truePlaceholder.id,
      falseBranch: falsePlaceholder.id,
    },
  }

  return { tokens, group }
}

/**
 * 深拷贝 Token（生成新 ID）
 */
export function cloneToken(token: FormulaToken): FormulaToken {
  return {
    ...JSON.parse(JSON.stringify(token)),
    id: uuidv4(),
    createdAt: Date.now(),
    templateGroupId: undefined, // 复制时脱离原模板组
  }
}

/**
 * 从子表达式创建 Token 序列
 */
export function expandSubExpression(subExpr: SubExpression): FormulaToken[] {
  return subExpr.tokens.map(t => cloneToken(t))
}
```

### 4.2 拖拽 Composable（核心逻辑）

```typescript
// ==================== composables/useDragAndDrop.ts ====================

import { ref, onMounted, onUnmounted } from 'vue'
import { useFormulaStore } from '@/stores/formulaStore'
import type { DragPayload, FormulaToken } from '@/types/formula'
import { createToken, createIfThenElseTemplate, createConstantToken, createVariableToken } from '@/utils/tokenFactory'

/**
 * 拖拽引擎 Composable
 */
export function useDragAndDrop() {
  const store = useFormulaStore()

  // ─── 拖拽状态 ───
  const isDragging = ref(false)
  const dragPayload = ref<DragPayload | null>(null)
  const dropIndicatorIndex = ref<number | null>(null)
  const ghostElement = ref<HTMLElement | null>(null)

  /**
   * ═══════════════════════════════════════
   * 1. Palette → Canvas 拖拽（从左侧拖入）
   * ═══════════════════════════════════════
   */

  /**
   * 开始从 Palette 拖拽
   */
  function handlePaletteDragStart(event: DragEvent, paletteItem: PaletteItem) {
    if (!event.dataTransfer) return

    const payload: DragPayload = {
      source: 'palette',
      tokenPrototype: {
        type: paletteItem.tokenType,
        category: paletteItem.category,
        displayText: paletteItem.label,
        outputValue: paletteItem.outputValue,
        editable: paletteItem.editable || false,
      },
      isTemplate: paletteItem.isTemplate || false,
      templateType: paletteItem.templateType,
    }

    event.dataTransfer.setData('application/formula-token', JSON.stringify(payload))
    event.dataTransfer.effectAllowed = 'copy'

    // 自定义拖拽幽灵
    const ghost = createDragGhost(paletteItem.label)
    document.body.appendChild(ghost)
    event.dataTransfer.setDragImage(ghost, ghost.offsetWidth / 2, ghost.offsetHeight / 2)
    setTimeout(() => document.body.removeChild(ghost), 0)

    isDragging.value = true
    dragPayload.value = payload
    store.dragState.isDragging = true
    store.dragState.payload = payload
  }

  /**
   * ═══════════════════════════════════════
   * 2. Canvas 内部拖拽（排序）
   * ═══════════════════════════════════════
   */

  function handleCanvasDragStart(event: DragEvent, token: FormulaToken, index: number) {
    if (!event.dataTransfer) return

    const payload: DragPayload = {
      source: 'canvas',
      tokenId: token.id,
    }

    event.dataTransfer.setData('application/formula-token', JSON.stringify(payload))
    event.dataTransfer.effectAllowed = 'move'

    // 设置拖拽元素半透明
    const el = event.currentTarget as HTMLElement
    setTimeout(() => el.classList.add('dragging'), 0)

    isDragging.value = true
    dragPayload.value = payload
    store.dragState.isDragging = true
    store.dragState.payload = payload
  }

  /**
   * ═══════════════════════════════════════
   * 3. Canvas DragOver 处理（计算插入位置）
   * ═══════════════════════════════════════
   */

  function handleCanvasDragOver(event: DragEvent, containerEl: HTMLElement) {
    event.preventDefault()
    event.dataTransfer!.dropEffect = dragPayload.value?.source === 'palette' ? 'copy' : 'move'

    // 计算最近的插入索引
    const insertIndex = calculateInsertIndex(event, containerEl)
    dropIndicatorIndex.value = insertIndex
    store.dragState.dropTargetIndex = insertIndex
  }

  /**
   * 计算鼠标位置对应的插入索引
   * 核心算法：遍历所有 Token DOM 元素，找到最近的间隙
   */
  function calculateInsertIndex(event: DragEvent, container: HTMLElement): number {
    const tokenElements = container.querySelectorAll<HTMLElement>('[data-token-index]')
    const mouseX = event.clientX
    const mouseY = event.clientY

    if (tokenElements.length === 0) return 0

    let closestIndex = 0
    let closestDistance = Infinity

    for (let i = 0; i < tokenElements.length; i++) {
      const rect = tokenElements[i].getBoundingClientRect()
      const tokenCenterX = rect.left + rect.width / 2

      // 判断鼠标在 Token 的左半还是右半
      if (mouseX < tokenCenterX) {
        // 在左侧 → 插入到该 Token 前面
        const distance = Math.abs(mouseX - rect.left)
        if (distance < closestDistance) {
          closestDistance = distance
          closestIndex = i
        }
      } else {
        // 在右侧 → 插入到该 Token 后面
        const distance = Math.abs(mouseX - rect.right)
        if (distance < closestDistance) {
          closestDistance = distance
          closestIndex = i + 1
        }
      }
    }

    // 如果是 canvas 内部移动，需要排除自身
    if (dragPayload.value?.source === 'canvas' && dragPayload.value.tokenId) {
      const sourceIndex = store.tokens.findIndex(t => t.id === dragPayload.value!.tokenId)
      if (sourceIndex !== -1 && closestIndex === sourceIndex) {
        return -1 // 无效位置（放回原位）
      }
    }

    return closestIndex
  }

  /**
   * ═══════════════════════════════════════
   * 4. Drop 处理
   * ═══════════════════════════════════════
   */

  function handleCanvasDrop(event: DragEvent) {
    event.preventDefault()
    const dataStr = event.dataTransfer?.getData('application/formula-token')
    if (!dataStr) return

    const payload: DragPayload = JSON.parse(dataStr)
    const insertIndex = dropIndicatorIndex.value ?? store.tokens.length

    if (payload.source === 'palette') {
      handleDropFromPalette(payload, insertIndex)
    } else if (payload.source === 'canvas') {
      handleDropFromCanvas(payload, insertIndex)
    }

    // 清理状态
    cleanupDragState()
  }

  /**
   * 处理从 Palette 拖入
   */
  function handleDropFromPalette(payload: DragPayload, index: number) {
    if (payload.isTemplate && payload.templateType === 'IF_THEN_ELSE') {
      // 展开 IF-THEN-ELSE 模板
      const { tokens: newTokens, group } = createIfThenElseTemplate()
      store.templateGroups.set(group.id, group)
      store.insertTokens(newTokens, index)
    } else if (payload.tokenPrototype) {
      const token = createToken(payload.tokenPrototype.type as TokenType, {
        displayText: payload.tokenPrototype.displayText,
        outputValue: payload.tokenPrototype.outputValue,
        editable: payload.tokenPrototype.editable,
      })
      store.insertToken(token, index)
    }
  }

  /**
   * 处理 Canvas 内部移动
   */
  function handleDropFromCanvas(payload: DragPayload, targetIndex: number) {
    if (!payload.tokenId) return
    const sourceIndex = store.tokens.findIndex(t => t.id === payload.tokenId)
    if (sourceIndex === -1 || targetIndex === -1) return
    if (sourceIndex === targetIndex) return

    store.moveToken(sourceIndex, targetIndex)
  }

  /**
   * ═══════════════════════════════════════
   * 5. DragEnd & DragLeave
   * ═══════════════════════════════════════
   */

  function handleDragEnd(event: DragEvent) {
    cleanupDragState()
    // 移除拖拽中样式
    document.querySelectorAll('.dragging').forEach(el => el.classList.remove('dragging'))
  }

  function handleCanvasDragLeave(event: DragEvent) {
    // 只在真正离开画布时清除指示器
    const container = event.currentTarget as HTMLElement
    const relatedTarget = event.relatedTarget as HTMLElement
    if (!container.contains(relatedTarget)) {
      dropIndicatorIndex.value = null
      store.dragState.dropTargetIndex = null
    }
  }

  function cleanupDragState() {
    isDragging.value = false
    dragPayload.value = null
    dropIndicatorIndex.value = null
    store.dragState = {
      isDragging: false,
      payload: null,
      dropTargetIndex: null,
      dropTargetTokenId: null,
    }
  }

  /**
   * 创建拖拽幽灵元素
   */
  function createDragGhost(text: string): HTMLElement {
    const ghost = document.createElement('div')
    ghost.textContent = text
    ghost.style.cssText = `
      position: absolute;
      top: -1000px;
      left: -1000px;
      padding: 6px 16px;
      background: #409eff;
      color: white;
      border-radius: 6px;
      font-size: 14px;
      font-weight: 500;
      box-shadow: 0 2px 12px rgba(0,0,0,0.15);
      white-space: nowrap;
    `
    return ghost
  }

  return {
    // 状态
    isDragging,
    dropIndicatorIndex,
    // Palette 事件
    handlePaletteDragStart,
    // Canvas 事件
    handleCanvasDragStart,
    handleCanvasDragOver,
    handleCanvasDragLeave,
    handleCanvasDrop,
    handleDragEnd,
  }
}
```

### 4.3 插入指示器与让位动画

```typescript
// ==================== composables/useDropIndicator.ts ====================

import { computed, watch, type Ref } from 'vue'
import { useFormulaStore } from '@/stores/formulaStore'

/**
 * 管理插入位置指示器与让位动画
 */
export function useDropIndicator(canvasRef: Ref<HTMLElement | null>) {
  const store = useFormulaStore()

  /** 指示线的位置 (px) */
  const indicatorPosition = computed(() => {
    if (store.dragState.dropTargetIndex === null || !canvasRef.value) return null

    const tokenElements = canvasRef.value.querySelectorAll<HTMLElement>('[data-token-index]')
    const index = store.dragState.dropTargetIndex

    if (tokenElements.length === 0) {
      return { x: 16, y: 16 } // 空画布时的默认位置
    }

    if (index >= tokenElements.length) {
      // 插入到末尾
      const lastEl = tokenElements[tokenElements.length - 1]
      const rect = lastEl.getBoundingClientRect()
      const containerRect = canvasRef.value.getBoundingClientRect()
      return {
        x: rect.right - containerRect.left + 4,
        y: rect.top - containerRect.top,
      }
    }

    // 插入到某个 Token 前面
    const targetEl = tokenElements[index]
    const rect = targetEl.getBoundingClientRect()
    const containerRect = canvasRef.value.getBoundingClientRect()
    return {
      x: rect.left - containerRect.left - 2,
      y: rect.top - containerRect.top,
    }
  })

  /**
   * 让位动画：当指示器位置变化时，给目标位置左右的 Token 添加 transform
   */
  watch(() => store.dragState.dropTargetIndex, (newIndex, oldIndex) => {
    if (!canvasRef.value) return
    const tokenElements = canvasRef.value.querySelectorAll<HTMLElement>('[data-token-index]')

    // 清除所有让位样式
    tokenElements.forEach(el => {
      el.style.transition = 'transform 0.2s ease'
      el.style.transform = ''
    })

    if (newIndex === null) return

    // 目标位置及之后的 Token 向右移动
    const GAP = 28 // 让出的间距
    for (let i = newIndex; i < tokenElements.length; i++) {
      tokenElements[i].style.transform = `translateX(${GAP}px)`
    }
  })

  return {
    indicatorPosition,
  }
}
```

### 4.5 dragover 节流优化

```typescript
// ==================== utils/throttle.ts ====================

/**
 * requestAnimationFrame 节流
 */
export function rafThrottle<T extends (...args: any[]) => void>(fn: T): T {
  let rafId: number | null = null
  let lastArgs: any[] | null = null

  const throttled = (...args: any[]) => {
    lastArgs = args
    if (rafId === null) {
      rafId = requestAnimationFrame(() => {
        fn(...lastArgs!)
        rafId = null
        lastArgs = null
      })
    }
  }
```

## 5. 校验引擎设计

### 5.1 校验规则体系

校验引擎采用**有限状态机（FSM）+ 规则链**的混合架构。Token 序列被视为一条有限状态链，每个 Token 根据其前驱/后继的类型组合进行合法性判定。

#### 5.1.1 Token 角色分类（用于校验）

```typescript
// ==================== engine/validator.ts ====================

/**
 * 校验角色 —— 将 Token 抽象为更粗粒度的角色
 * 用于判断相邻两个 Token 是否合法
 */
export enum TokenRole {
  VALUE          = 'VALUE',           // 常量、变量、子表达式
  BINARY_OP      = 'BINARY_OP',       // 二元运算符：+ - * / && || == != > < >= <=
  UNARY_PREFIX   = 'UNARY_PREFIX',    // 前缀一元运算符：NOT (!)
  LPAREN         = 'LPAREN',          // (
  RPAREN         = 'RPAREN',          // )
  IF             = 'IF',              // 如果
  THEN           = 'THEN',            // 那么
  ELSE           = 'ELSE',            // 否则
  PLACEHOLDER    = 'PLACEHOLDER',     // 占位符
}

/** Token 类型 → 校验角色 映射 */
export function getTokenRole(type: TokenType): TokenRole {
  switch (type) {
    case TokenType.CONSTANT:
    case TokenType.VARIABLE:
    case TokenType.SUB_EXPR:
      return TokenRole.VALUE

    case TokenType.ADD: case TokenType.SUB: case TokenType.MUL:
    case TokenType.DIV: case TokenType.MOD:
    case TokenType.AND: case TokenType.OR:
    case TokenType.EQ:  case TokenType.NEQ:
    case TokenType.GT:  case TokenType.LT:
    case TokenType.GTE: case TokenType.LTE:
      return TokenRole.BINARY_OP

    case TokenType.NOT:
      return TokenRole.UNARY_PREFIX

    case TokenType.LPAREN:
      return TokenRole.LPAREN

    case TokenType.RPAREN:
      return TokenRole.RPAREN

    case TokenType.IF:
      return TokenRole.IF

    case TokenType.THEN:
      return TokenRole.THEN

    case TokenType.ELSE:
      return TokenRole.ELSE

    case TokenType.PLACEHOLDER:
      return TokenRole.PLACEHOLDER

    default:
      return TokenRole.VALUE
  }
}
```

#### 5.1.2 相邻合法性矩阵

定义一个**左→右合法性矩阵**：给定左侧 Token 角色和右侧 Token 角色，是否合法。

```typescript
/**
 * 相邻 Token 合法性矩阵
 * adjacencyMatrix[leftRole][rightRole] = true 表示合法
 *
 * 特殊约定：
 * - START 表示序列开头（左侧为空）
 * - END 表示序列结尾（右侧为空）
 */

type AdjacencyKey = TokenRole | 'START' | 'END'

const adjacencyMatrix: Record<AdjacencyKey, Set<AdjacencyKey>> = {
  // 序列开头之后可以出现：
  'START': new Set([
    TokenRole.VALUE,
    TokenRole.UNARY_PREFIX,
    TokenRole.LPAREN,
    TokenRole.IF,
    TokenRole.PLACEHOLDER,
  ]),

  // VALUE 之后可以出现：
  [TokenRole.VALUE]: new Set([
    TokenRole.BINARY_OP,
    TokenRole.RPAREN,
    TokenRole.THEN,
    TokenRole.ELSE,
    'END',
  ]),

  // BINARY_OP 之后可以出现：
  [TokenRole.BINARY_OP]: new Set([
    TokenRole.VALUE,
    TokenRole.UNARY_PREFIX,
    TokenRole.LPAREN,
    TokenRole.IF,
    TokenRole.PLACEHOLDER,
  ]),

  // UNARY_PREFIX (NOT) 之后可以出现：
  [TokenRole.UNARY_PREFIX]: new Set([
    TokenRole.VALUE,
    TokenRole.LPAREN,
    TokenRole.UNARY_PREFIX, // !!x 虽然少见但语法上合法
    TokenRole.PLACEHOLDER,
  ]),

  // LPAREN 之后可以出现：
  [TokenRole.LPAREN]: new Set([
    TokenRole.VALUE,
    TokenRole.UNARY_PREFIX,
    TokenRole.LPAREN,
    TokenRole.IF,
    TokenRole.PLACEHOLDER,
  ]),

  // RPAREN 之后可以出现：
  [TokenRole.RPAREN]: new Set([
    TokenRole.BINARY_OP,
    TokenRole.RPAREN,
    TokenRole.THEN,
    TokenRole.ELSE,
    'END',
  ]),

  // IF 之后可以出现：
  [TokenRole.IF]: new Set([
    TokenRole.VALUE,
    TokenRole.UNARY_PREFIX,
    TokenRole.LPAREN,
    TokenRole.PLACEHOLDER,
  ]),

  // THEN 之后可以出现：
  [TokenRole.THEN]: new Set([
    TokenRole.VALUE,
    TokenRole.UNARY_PREFIX,
    TokenRole.LPAREN,
    TokenRole.IF,
    TokenRole.PLACEHOLDER,
  ]),

  // ELSE 之后可以出现：
  [TokenRole.ELSE]: new Set([
    TokenRole.VALUE,
    TokenRole.UNARY_PREFIX,
    TokenRole.LPAREN,
    TokenRole.IF,
    TokenRole.PLACEHOLDER,
  ]),

  // PLACEHOLDER 之后（视情况，占位符本质上代表"待填"）
  [TokenRole.PLACEHOLDER]: new Set([
    TokenRole.BINARY_OP,
    TokenRole.RPAREN,
    TokenRole.THEN,
    TokenRole.ELSE,
    'END',
  ]),

  // 序列结尾（不需要，用于标记哪些角色可以作为结尾）
  'END': new Set([]),
}

/**
 * 可以出现在序列末尾的角色
 */
const VALID_END_ROLES = new Set<TokenRole>([
  TokenRole.VALUE,
  TokenRole.RPAREN,
  TokenRole.PLACEHOLDER, // 如果还有占位符，虽然不完整但暂不报错
])
```

#### 5.1.3 校验引擎主函数

```typescript
/** 校验错误 */
export interface ValidationError {
  tokenId: string
  tokenIndex: number
  message: string
  severity: 'error' | 'warning'
  /** 关联 Token（如两个常量之间缺运算符，两个都标记） */
  relatedTokenIds?: string[]
}

/**
 * 主校验函数 —— 对 Token 序列进行完整校验
 */
export function validateTokenSequence(tokens: FormulaToken[]): ValidationError[] {
  const errors: ValidationError[] = []

  if (tokens.length === 0) return errors

  // ─── 1. 相邻合法性校验 ───
  const adjacencyErrors = validateAdjacency(tokens)
  errors.push(...adjacencyErrors)

  // ─── 2. 括号匹配校验 ───
  const bracketErrors = validateBrackets(tokens)
  errors.push(...bracketErrors)

  // ─── 3. IF-THEN-ELSE 完整性校验 ───
  const controlFlowErrors = validateControlFlow(tokens)
  errors.push(...controlFlowErrors)

  // ─── 4. 占位符未填写校验（警告级别） ───
  const placeholderWarnings = validatePlaceholders(tokens)
  errors.push(...placeholderWarnings)

  // ─── 5. 空值校验（可编辑 Token 值为空） ───
  const emptyValueErrors = validateEmptyValues(tokens)
  errors.push(...emptyValueErrors)

  // ─── 更新 Token 的校验状态 ───
  applyValidationResults(tokens, errors)

  return errors
}

/**
 * 相邻合法性校验
 */
function validateAdjacency(tokens: FormulaToken[]): ValidationError[] {
  const errors: ValidationError[] = []
  const nonPlaceholderTokens = tokens // 包含占位符一起校验

  for (let i = 0; i < nonPlaceholderTokens.length; i++) {
    const current = nonPlaceholderTokens[i]
    const currentRole = getTokenRole(current.type)
    const prevRole: AdjacencyKey = i === 0 ? 'START' : getTokenRole(nonPlaceholderTokens[i - 1].type)

    // 检查 prev → current 是否合法
    if (!adjacencyMatrix[prevRole]?.has(currentRole)) {
      const prevToken = i > 0 ? nonPlaceholderTokens[i - 1] : null
      const errMsg = generateAdjacencyErrorMessage(prevRole, currentRole, prevToken, current)

      errors.push({
        tokenId: current.id,
        tokenIndex: i,
        message: errMsg,
        severity: 'error',
        relatedTokenIds: prevToken ? [prevToken.id, current.id] : [current.id],
      })
    }

    // 检查最后一个 Token 是否可以作为结尾
    if (i === nonPlaceholderTokens.length - 1) {
      if (!VALID_END_ROLES.has(currentRole) && currentRole !== TokenRole.PLACEHOLDER) {
        errors.push({
          tokenId: current.id,
          tokenIndex: i,
          message: `表达式不能以「${current.displayText}」结尾`,
          severity: 'error',
        })
      }
    }
  }

  return errors
}

/**
 * 生成友好的相邻错误提示信息
 */
function generateAdjacencyErrorMessage(
  leftRole: AdjacencyKey,
  rightRole: AdjacencyKey,
  leftToken: FormulaToken | null,
  rightToken: FormulaToken
): string {
  // 常见情况的友好提示
  if (leftRole === TokenRole.VALUE && rightRole === TokenRole.VALUE) {
    return `「${leftToken?.displayText}」和「${rightToken.displayText}」之间缺少运算符`
  }
  if (leftRole === TokenRole.BINARY_OP && rightRole === TokenRole.BINARY_OP) {
    return `不能连续使用两个运算符「${leftToken?.displayText}」「${rightToken.displayText}」`
  }
  if (leftRole === 'START' && rightRole === TokenRole.BINARY_OP) {
    return `表达式不能以运算符「${rightToken.displayText}」开头`
  }
  if (leftRole === TokenRole.BINARY_OP && rightRole === 'END') {
    return `运算符「${leftToken?.displayText}」后面缺少操作数`
  }
  // 默认提示
  return `「${leftToken?.displayText || '开头'}」后面不能直接跟「${rightToken.displayText}」`
}

/**
 * 括号匹配校验
 */
function validateBrackets(tokens: FormulaToken[]): ValidationError[] {
  const errors: ValidationError[] = []
  const stack: Array<{ token: FormulaToken; index: number }> = []

  for (let i = 0; i < tokens.length; i++) {
    const token = tokens[i]
    if (token.type === TokenType.LPAREN) {
      stack.push({ token, index: i })
    } else if (token.type === TokenType.RPAREN) {
      if (stack.length === 0) {
        errors.push({
          tokenId: token.id,
          tokenIndex: i,
          message: '多余的右括号，没有匹配的左括号',
          severity: 'error',
        })
      } else {
        stack.pop()
      }
    }
  }

  // 未闭合的左括号
  for (const { token, index } of stack) {
    errors.push({
      tokenId: token.id,
      tokenIndex: index,
      message: '左括号没有匹配的右括号',
      severity: 'error',
    })
  }

  return errors
}

/**
 * IF-THEN-ELSE 结构完整性校验
 */
function validateControlFlow(tokens: FormulaToken[]): ValidationError[] {
  const errors: ValidationError[] = []

  // 收集所有模板组
  const groupMap = new Map<string, FormulaToken[]>()
  for (const token of tokens) {
    if (token.templateGroupId) {
      if (!groupMap.has(token.templateGroupId)) {
        groupMap.set(token.templateGroupId, [])
      }
      groupMap.get(token.templateGroupId)!.push(token)
    }
  }

  // 检查每个模板组的完整性
  for (const [groupId, groupTokens] of groupMap) {
    const hasIf = groupTokens.some(t => t.type === TokenType.IF)
    const hasThen = groupTokens.some(t => t.type === TokenType.THEN)
    const hasElse = groupTokens.some(t => t.type === TokenType.ELSE)

    if (!hasIf || !hasThen || !hasElse) {
      const missing = []
      if (!hasIf) missing.push('如果')
      if (!hasThen) missing.push('那么')
      if (!hasElse) missing.push('否则')

      for (const token of groupTokens) {
        if ([TokenType.IF, TokenType.THEN, TokenType.ELSE].includes(token.type)) {
          errors.push({
            tokenId: token.id,
            tokenIndex: tokens.indexOf(token),
            message: `条件表达式不完整，缺少：${missing.join('、')}`,
            severity: 'error',
          })
        }
      }
    }

    // 检查顺序：IF 必须在 THEN 前面，THEN 必须在 ELSE 前面
    const ifIndex = tokens.findIndex(t => t.templateGroupId === groupId && t.type === TokenType.IF)
    const thenIndex = tokens.findIndex(t => t.templateGroupId === groupId && t.type === TokenType.THEN)
    const elseIndex = tokens.findIndex(t => t.templateGroupId === groupId && t.type === TokenType.ELSE)

    if (ifIndex !== -1 && thenIndex !== -1 && ifIndex > thenIndex) {
      errors.push({
        tokenId: groupTokens.find(t => t.type === TokenType.IF)!.id,
        tokenIndex: ifIndex,
        message: '「如果」必须在「那么」之前',
        severity: 'error',
      })
    }
    if (thenIndex !== -1 && elseIndex !== -1 && thenIndex > elseIndex) {
      errors.push({
        tokenId: groupTokens.find(t => t.type === TokenType.THEN)!.id,
        tokenIndex: thenIndex,
        message: '「那么」必须在「否则」之前',
        severity: 'error',
      })
    }
  }

  // 检查孤立的 THEN/ELSE（不属于任何模板组）
  for (let i = 0; i < tokens.length; i++) {
    const token = tokens[i]
    if ((token.type === TokenType.THEN || token.type === TokenType.ELSE) && !token.templateGroupId) {
      errors.push({
        tokenId: token.id,
        tokenIndex: i,
        message: `孤立的「${token.displayText}」，缺少配对的「如果」`,
        severity: 'error',
      })
    }
  }

  return errors
}

/**
 * 占位符未填写检查
 */
function validatePlaceholders(tokens: FormulaToken[]): ValidationError[] {
  return tokens
    .filter(t => t.type === TokenType.PLACEHOLDER)
    .map((t, _, arr) => ({
      tokenId: t.id,
      tokenIndex: tokens.indexOf(t),
      message: `请填写${t.displayText || '此处'}`,
      severity: 'warning' as const,
    }))
}

/**
 * 可编辑 Token 空值检查
 */
function validateEmptyValues(tokens: FormulaToken[]): ValidationError[] {
  return tokens
    .filter(t => t.editable && (!t.displayText || t.displayText.trim() === ''))
    .map(t => ({
      tokenId: t.id,
      tokenIndex: tokens.indexOf(t),
      message: '值不能为空',
      severity: 'error' as const,
    }))
}

/**
 * 将校验结果应用到 Token 上
 */
function applyValidationResults(tokens: FormulaToken[], errors: ValidationError[]) {
  // 先重置所有状态
  for (const token of tokens) {
    token.validationState = 'valid'
    token.validationMessage = undefined
  }

  // 应用错误
  for (const error of errors) {
    const relatedIds = error.relatedTokenIds || [error.tokenId]
    for (const id of relatedIds) {
      const token = tokens.find(t => t.id === id)
      if (token) {
        // error 优先级高于 warning
        if (error.severity === 'error' || token.validationState !== 'error') {
          token.validationState = error.severity
          token.validationMessage = error.message
        }
      }
    }
  }
}
```

## 8. 编译器引擎（Token → 后端表达式）

### 8.1 编译架构概览

编译过程分两条链路，都从同一份 Token 序列出发：

```
Token[]
  │
  ├──▶ 中文可读链路：Token.displayText 直接拼接 → "如果(a==100 并且 ...) 那么 ..."
  │
  └──▶ 后端表达式链路：Token[] → AST → CodeGen → "a==100&&(...)?...:..."
                           ▲         ▲
                           │         │
                       解析阶段    生成阶段
```

为什么需要 AST 中间层而不是直接拼接 `outputValue`：

1. IF/THEN/ELSE 需要转换为三元运算符 `? :`，结构不是简单拼接
2. 嵌套 IF 需要正确添加括号
3. 子表达式展开时需要加括号保证优先级
4. 将来可扩展为其他目标语言（如 SQL、Python）

### 8.2 AST 节点定义

```typescript
// ==================== engine/formula/ast.ts ====================

/**
 * 抽象语法树节点类型
 */
export type ASTNode =
  | LiteralNode
  | VariableNode
  | BinaryExprNode
  | UnaryExprNode
  | ConditionalExprNode
  | GroupExprNode
  | SubExprRefNode
  | PlaceholderNode

/** 字面量（常量） */
export interface LiteralNode {
  kind: 'Literal'
  value: string         // 原始值
  rawDisplay: string    // 显示文本
  tokenId: string
}

/** 变量引用 */
export interface VariableNode {
  kind: 'Variable'
  name: string          // 变量名（输出用）
  displayName: string   // 显示名
  tokenId: string
}

/** 二元运算 */
export interface BinaryExprNode {
  kind: 'BinaryExpr'
  operator: string      // 输出运算符：+, -, *, /, &&, ||, ==, !=, >, <, >=, <=
  displayOp: string     // 显示运算符
  left: ASTNode
  right: ASTNode
}

/** 一元运算（前缀） */
export interface UnaryExprNode {
  kind: 'UnaryExpr'
  operator: string      // !
  displayOp: string     // 非
  operand: ASTNode
}

/** 条件表达式 (IF-THEN-ELSE → 三元) */
export interface ConditionalExprNode {
  kind: 'ConditionalExpr'
  condition: ASTNode
  consequent: ASTNode   // THEN 分支
  alternate: ASTNode    // ELSE 分支
}

/** 分组表达式（括号） */
export interface GroupExprNode {
  kind: 'GroupExpr'
  expression: ASTNode
}

/** 子表达式引用 */
export interface SubExprRefNode {
  kind: 'SubExprRef'
  subExprId: string
  subExprName: string
  expanded: ASTNode     // 展开后的 AST
}

/** 占位符（编译时视为错误或占位） */
export interface PlaceholderNode {
  kind: 'Placeholder'
  label: string
  tokenId: string
}
```

### 8.3 Parser —— Token 序列 → AST

采用 **递归下降** 解析，运算符优先级从低到高：

```
优先级（从低到高）：
  1. 条件表达式  IF...THEN...ELSE
  2. 逻辑或      OR  (||)
  3. 逻辑与      AND (&&)
  4. 比较        ==, !=, >, <, >=, <=
  5. 加减        +, -
  6. 乘除取余    *, /, %
  7. 一元前缀    NOT (!)
  8. 原子        常量, 变量, 括号组, 子表达式
```

```typescript
// ==================== engine/formula/parser.ts ====================

import type { FormulaToken } from '@/types/formula'
import { TokenType } from '@/types/formula'
import type { ASTNode } from './ast'

/**
 * 递归下降 Parser
 * 将 Token 序列解析为 AST
 */
export class FormulaParser {
  private tokens: FormulaToken[]
  private pos: number = 0
  private errors: string[] = []

  constructor(tokens: FormulaToken[]) {
    // 过滤掉占位符，将其保留为特殊节点
    this.tokens = tokens
  }

  /** 入口 */
  parse(): { ast: ASTNode | null; errors: string[] } {
    this.pos = 0
    this.errors = []

    if (this.tokens.length === 0) {
      return { ast: null, errors: ['空表达式'] }
    }

    try {
      const ast = this.parseExpression()

      if (this.pos < this.tokens.length) {
        this.errors.push(`意外的 Token: "${this.peek()?.displayText}" 位置 ${this.pos}`)
      }

      return { ast, errors: this.errors }
    } catch (e: any) {
      this.errors.push(e.message || '解析错误')
      return { ast: null, errors: this.errors }
    }
  }

  // ─── 辅助方法 ───

  private peek(): FormulaToken | null {
    return this.pos < this.tokens.length ? this.tokens[this.pos] : null
  }

  private advance(): FormulaToken {
    const token = this.tokens[this.pos]
    this.pos++
    return token
  }

  private expect(type: TokenType): FormulaToken {
    const token = this.peek()
    if (!token || token.type !== type) {
      throw new Error(`期望 ${type}，但得到 ${token?.type || 'EOF'}`)
    }
    return this.advance()
  }

  private match(...types: TokenType[]): boolean {
    const token = this.peek()
    return token !== null && types.includes(token.type)
  }

  // ─── 解析规则（优先级从低到高） ───

  /** 顶层表达式 = 条件表达式 */
  private parseExpression(): ASTNode {
    return this.parseConditional()
  }

  /**
   * 条件表达式: IF expr THEN expr ELSE expr
   * 如果没有 IF 开头，则降级为 logicalOr
   */
  private parseConditional(): ASTNode {
    if (this.match(TokenType.IF)) {
      this.advance() // 消费 IF

      const condition = this.parseLogicalOr()

      // 期望 THEN
      if (!this.match(TokenType.THEN)) {
        this.errors.push('缺少「那么」')
        return this.makePlaceholder('missing-then')
      }
      this.advance() // 消费 THEN

      const consequent = this.parseExpression() // 递归：THEN 分支可以包含嵌套 IF

      // 期望 ELSE
      if (!this.match(TokenType.ELSE)) {
        this.errors.push('缺少「否则」')
        return this.makePlaceholder('missing-else')
      }
      this.advance() // 消费 ELSE

      const alternate = this.parseExpression() // 递归：ELSE 分支也可以嵌套 IF

      return {
        kind: 'ConditionalExpr',
        condition,
        consequent,
        alternate,
      }
    }

    return this.parseLogicalOr()
  }

  /** 逻辑或: logicalAnd (OR logicalAnd)* */
  private parseLogicalOr(): ASTNode {
    let left = this.parseLogicalAnd()

    while (this.match(TokenType.OR)) {
      const opToken = this.advance()
      const right = this.parseLogicalAnd()
      left = {
        kind: 'BinaryExpr',
        operator: '||',
        displayOp: '或者',
        left,
        right,
      }
    }

    return left
  }

  /** 逻辑与: comparison (AND comparison)* */
  private parseLogicalAnd(): ASTNode {
    let left = this.parseComparison()

    while (this.match(TokenType.AND)) {
      const opToken = this.advance()
      const right = this.parseComparison()
      left = {
        kind: 'BinaryExpr',
        operator: '&&',
        displayOp: '并且',
        left,
        right,
      }
    }

    return left
  }

  /** 比较: addition ((== | != | > | < | >= | <=) addition)* */
  private parseComparison(): ASTNode {
    let left = this.parseAddition()

    const compOps = [
      TokenType.EQ, TokenType.NEQ,
      TokenType.GT, TokenType.LT,
      TokenType.GTE, TokenType.LTE,
    ]

    while (this.match(...compOps)) {
      const opToken = this.advance()
      const right = this.parseAddition()
      left = {
        kind: 'BinaryExpr',
        operator: opToken.outputValue,
        displayOp: opToken.displayText,
        left,
        right,
      }
    }

    return left
  }

  /** 加减: multiplication ((+ | -) multiplication)* */
  private parseAddition(): ASTNode {
    let left = this.parseMultiplication()

    while (this.match(TokenType.ADD, TokenType.SUB)) {
      const opToken = this.advance()
      const right = this.parseMultiplication()
      left = {
        kind: 'BinaryExpr',
        operator: opToken.outputValue,
        displayOp: opToken.displayText,
        left,
        right,
      }
    }

    return left
  }

  /** 乘除取余: unary ((* | / | %) unary)* */
  private parseMultiplication(): ASTNode {
    let left = this.parseUnary()

    while (this.match(TokenType.MUL, TokenType.DIV, TokenType.MOD)) {
      const opToken = this.advance()
      const right = this.parseUnary()
      left = {
        kind: 'BinaryExpr',
        operator: opToken.outputValue,
        displayOp: opToken.displayText,
        left,
        right,
      }
    }

    return left
  }

  /** 一元前缀: NOT? atom */
  private parseUnary(): ASTNode {
    if (this.match(TokenType.NOT)) {
      const opToken = this.advance()
      const operand = this.parseUnary() // 递归支持 !!x
      return {
        kind: 'UnaryExpr',
        operator: '!',
        displayOp: '非',
        operand,
      }
    }

    return this.parseAtom()
  }

  /** 原子: 常量 | 变量 | 子表达式 | 括号组 | 占位符 */
  private parseAtom(): ASTNode {
    const token = this.peek()

    if (!token) {
      this.errors.push('表达式意外结束')
      return this.makePlaceholder('unexpected-end')
    }

    // 常量
    if (token.type === TokenType.CONSTANT) {
      this.advance()
      return {
        kind: 'Literal',
        value: token.outputValue,
        rawDisplay: token.displayText,
        tokenId: token.id,
      }
    }

    // 变量
    if (token.type === TokenType.VARIABLE) {
      this.advance()
      return {
        kind: 'Variable',
        name: token.outputValue,
        displayName: token.displayText,
        tokenId: token.id,
      }
    }

    // 子表达式
    if (token.type === TokenType.SUB_EXPR) {
      this.advance()
      // 展开子表达式的 Token 并递归解析
      if (token.subExprTokens && token.subExprTokens.length > 0) {
        const subParser = new FormulaParser(token.subExprTokens)
        const { ast, errors } = subParser.parse()
        this.errors.push(...errors.map(e => `子表达式「${token.displayText}」: ${e}`))
        return {
          kind: 'SubExprRef',
          subExprId: token.subExprId || '',
          subExprName: token.displayText,
          expanded: ast || this.makePlaceholder('sub-expr-error'),
        }
      }
      return this.makePlaceholder('empty-sub-expr')
    }

    // 左括号 → 括号组
    if (token.type === TokenType.LPAREN) {
      this.advance() // 消费 (
      const expr = this.parseExpression()

      if (!this.match(TokenType.RPAREN)) {
        this.errors.push('缺少右括号')
      } else {
        this.advance() // 消费 )
      }

      return {
        kind: 'GroupExpr',
        expression: expr,
      }
    }

    // 占位符
    if (token.type === TokenType.PLACEHOLDER) {
      this.advance()
      return {
        kind: 'Placeholder',
        label: token.displayText,
        tokenId: token.id,
      }
    }

    // 意外 Token
    this.errors.push(`意外的「${token.displayText}」`)
    this.advance() // 跳过以避免无限循环
    return this.makePlaceholder('unexpected-' + token.type)
  }

  private makePlaceholder(label: string): PlaceholderNode {
    return { kind: 'Placeholder', label, tokenId: '' }
  }
}
```

### 8.4 CodeGenerator —— AST → 目标表达式

```typescript
// ==================== engine/formula/codeGenerator.ts ====================

import type { ASTNode } from './ast'

/**
 * 代码生成器
 * AST → 后端标准表达式字符串
 */
export class CodeGenerator {
  /**
   * 生成后端表达式
   */
  generateExpression(ast: ASTNode): string {
    return this.visitNode(ast)
  }

  /**
   * 生成中文可读表达式
   */
  generateDisplay(ast: ASTNode): string {
    return this.visitNodeDisplay(ast)
  }

  // ═══════════════ 后端表达式生成 ═══════════════

  private visitNode(node: ASTNode): string {
    switch (node.kind) {
      case 'Literal':
        return this.genLiteral(node)
      case 'Variable':
        return node.name
      case 'BinaryExpr':
        return this.genBinary(node)
      case 'UnaryExpr':
        return this.genUnary(node)
      case 'ConditionalExpr':
        return this.genConditional(node)
      case 'GroupExpr':
        return `(${this.visitNode(node.expression)})`
      case 'SubExprRef':
        // 子表达式展开并加括号
        return `(${this.visitNode(node.expanded)})`
      case 'Placeholder':
        return `__PLACEHOLDER__` // 标记未填写位置
      default:
        return ''
    }
  }

  private genLiteral(node: LiteralNode): string {
    const val = node.value.trim()

    // 判断是否为数字
    if (/^-?\d+(\.\d+)?$/.test(val)) {
      return val
    }

    // 判断是否为布尔值
    if (val === 'true' || val === 'false') {
      return val
    }

    // 字符串值：判断是否需要加引号
    // 如果看起来像变量名（纯字母数字下划线），不加引号
    if (/^[a-zA-Z_\u4e00-\u9fa5][\w\u4e00-\u9fa5]*$/.test(val)) {
      return val
    }

    // 其他情况加引号
    return `"${val.replace(/"/g, '\\"')}"`
  }

  private genBinary(node: BinaryExprNode): string {
    const left = this.visitNode(node.left)
    const right = this.visitNode(node.right)

    // 判断是否需要给子表达式加括号（优先级处理）
    const leftStr = this.maybeParenthesize(node.left, node, 'left')
    const rightStr = this.maybeParenthesize(node.right, node, 'right')

    return `${leftStr}${node.operator}${rightStr}`
  }

  private genUnary(node: UnaryExprNode): string {
    const operand = this.visitNode(node.operand)
    // NOT 后面如果是复杂表达式需要括号
    if (node.operand.kind === 'BinaryExpr' || node.operand.kind === 'ConditionalExpr') {
      return `${node.operator}(${operand})`
    }
    return `${node.operator}${operand}`
  }

  private genConditional(node: ConditionalExprNode): string {
    const condition = this.visitNode(node.condition)
    const consequent = this.visitNode(node.consequent)
    const alternate = this.visitNode(node.alternate)

    // 嵌套的条件表达式要加括号
    const condStr = node.condition.kind === 'ConditionalExpr'
      ? `(${condition})`
      : condition

    return `${condStr}?${consequent}:${alternate}`
  }

  /**
   * 根据运算符优先级决定是否加括号
   */
  private maybeParenthesize(
    child: ASTNode,
    parent: BinaryExprNode,
    side: 'left' | 'right'
  ): string {
    const childStr = this.visitNode(child)

    if (child.kind !== 'BinaryExpr') return childStr

    const childPriority = this.getOperatorPriority(child.operator)
    const parentPriority = this.getOperatorPriority(parent.operator)

    // 子节点优先级更低 → 需要括号
    if (childPriority < parentPriority) {
      return `(${childStr})`
    }

    // 同优先级，右侧需要括号（避免歧义，如 a-(b-c)）
    if (childPriority === parentPriority && side === 'right') {
      const nonAssociative = ['-', '/', '%']
      if (nonAssociative.includes(parent.operator)) {
        return `(${childStr})`
      }
    }

    return childStr
  }

  private getOperatorPriority(op: string): number {
    const priorities: Record<string, number> = {
      '||': 1,
      '&&': 2,
      '==': 3, '!=': 3, '>': 3, '<': 3, '>=': 3, '<=': 3,
      '+': 4, '-': 4,
      '*': 5, '/': 5, '%': 5,
    }
    return priorities[op] || 0
  }

  // ═══════════════ 中文可读表达式生成 ═══════════════

  private visitNodeDisplay(node: ASTNode): string {
    switch (node.kind) {
      case 'Literal':
        return node.rawDisplay
      case 'Variable':
        return node.displayName
      case 'BinaryExpr':
        return `${this.visitNodeDisplay(node.left)} ${node.displayOp} ${this.visitNodeDisplay(node.right)}`
      case 'UnaryExpr':
        return `${node.displayOp}(${this.visitNodeDisplay(node.operand)})`
      case 'ConditionalExpr':
        return `如果(${this.visitNodeDisplay(node.condition)}) 那么 ${this.visitNodeDisplay(node.consequent)} 否则 ${this.visitNodeDisplay(node.alternate)}`
      case 'GroupExpr':
        return `(${this.visitNodeDisplay(node.expression)})`
      case 'SubExprRef':
        return `[${node.subExprName}]`
      case 'Placeholder':
        return `___`
      default:
        return ''
    }
  }
}
```

### 8.5 编译器 Composable 封装

```typescript
// ==================== composables/formula/useFormulaCompiler.ts ====================

import { computed, watch, ref } from 'vue'
import { useFormulaStore } from '@/stores/formulaStore'
import { FormulaParser } from '@/engine/formula/parser'
import { CodeGenerator } from '@/engine/formula/codeGenerator'
import type { ASTNode } from '@/engine/formula/ast'

export function useFormulaCompiler() {
  const store = useFormulaStore()
  const generator = new CodeGenerator()

  const parseErrors = ref<string[]>([])
  const currentAST = ref<ASTNode | null>(null)

  /**
   * 后端标准表达式
   */
  const compiledExpression = computed(() => {
    if (!currentAST.value) return ''
    try {
      return generator.generateExpression(currentAST.value)
    } catch {
      return '// 编译错误'
    }
  })

  /**
   * 中文可读表达式
   */
  const displayExpression = computed(() => {
    if (!currentAST.value) return ''
    try {
      return generator.generateDisplay(currentAST.value)
    } catch {
      return '// 编译错误'
    }
  })

  /**
   * 简单拼接模式 —— 不经过 AST，直接拼 outputValue
   * 用于快速预览，当 AST 解析失败时的 fallback
   */
  const rawExpression = computed(() => {
    return store.tokens
      .filter(t => t.type !== 'PLACEHOLDER')
      .map(t => t.outputValue)
      .join('')
  })

  /**
   * 简单拼接的中文表达
   */
  const rawDisplayExpression = computed(() => {
    return store.tokens
      .filter(t => t.type !== 'PLACEHOLDER')
      .map(t => t.displayText)
      .join(' ')
  })

  /**
   * 是否编译成功（无解析错误且无占位符）
   */
  const isCompileSuccess = computed(() => {
    return parseErrors.value.length === 0
      && currentAST.value !== null
      && !hasPlaceholderInAST(currentAST.value)
  })

  /**
   * 监听 Token 变化，自动重新解析
   */
  watch(
    () => [...store.tokens],
    (newTokens) => {
      if (newTokens.length === 0) {
        currentAST.value = null
        parseErrors.value = []
        return
      }

      const parser = new FormulaParser(newTokens)
      const { ast, errors } = parser.parse()
      currentAST.value = ast
      parseErrors.value = errors
    },
    { deep: true, immediate: true }
  )

  /**
   * 检查 AST 中是否还有占位符节点
   */
  function hasPlaceholderInAST(node: ASTNode): boolean {
    if (node.kind === 'Placeholder') return true
    if (node.kind === 'BinaryExpr') {
      return hasPlaceholderInAST(node.left) || hasPlaceholderInAST(node.right)
    }
    if (node.kind === 'UnaryExpr') return hasPlaceholderInAST(node.operand)
    if (node.kind === 'ConditionalExpr') {
      return hasPlaceholderInAST(node.condition)
        || hasPlaceholderInAST(node.consequent)
        || hasPlaceholderInAST(node.alternate)
    }
    if (node.kind === 'GroupExpr') return hasPlaceholderInAST(node.expression)
    if (node.kind === 'SubExprRef') return hasPlaceholderInAST(node.expanded)
    return false
  }

  /**
   * 手动触发完整编译（供外部保存时调用）
   */
  function compile(): {
    success: boolean
    expression: string
    display: string
    errors: string[]
  } {
    if (store.tokens.length === 0) {
      return { success: false, expression: '', display: '', errors: ['空表达式'] }
    }

    const parser = new FormulaParser(store.tokens)
    const { ast, errors } = parser.parse()

    if (!ast || errors.length > 0) {
      return { success: false, expression: '', display: '', errors }
    }

    if (hasPlaceholderInAST(ast)) {
      return {
        success: false,
        expression: '',
        display: '',
        errors: ['存在未填写的占位符'],
      }
    }

    return {
      success: true,
      expression: generator.generateExpression(ast),
      display: generator.generateDisplay(ast),
      errors: [],
    }
  }

  return {
    compiledExpression,
    displayExpression,
    rawExpression,
    rawDisplayExpression,
    parseErrors,
    currentAST,
    isCompileSuccess,
    compile,
  }
}
```

```typescript
// ==================== stores/formulaStore.ts（补充 subExpression 相关方法） ====================

/** 添加子表达式 */
function addSubExpression(
  name: string,
  description: string,
  tokenSlice: FormulaToken[]
): { success: boolean; error?: string } {
  // 名称唯一性
  if (subExpressions.value.some(s => s.name === name)) {
    return { success: false, error: '子表达式名称已存在' }
  }

  // 名称合法性
  if (!name.trim()) {
    return { success: false, error: '名称不能为空' }
  }

  // 表达式完整性校验
  const validationErrors = validateTokenSequence([...tokenSlice])
  const hasRealErrors = validationErrors.some(e => e.severity === 'error')
  if (hasRealErrors) {
    return { success: false, error: '所选表达式存在语法错误，请先修正' }
  }

  // 占位符检查
  const hasPlaceholders = tokenSlice.some(t => t.type === TokenType.PLACEHOLDER)
  if (hasPlaceholders) {
    return { success: false, error: '所选表达式包含未填写的占位符' }
  }

  // 深拷贝并脱离模板组
  const clonedTokens = tokenSlice.map(t => ({
    ...JSON.parse(JSON.stringify(t)),
    id: uuidv4(),
    templateGroupId: undefined,
    validationState: 'valid' as const,
  }))

  // 编译
  const parser = new FormulaParser(clonedTokens)
  const { ast } = parser.parse()
  const generator = new CodeGenerator()
  const compiledExpr = ast ? generator.generateExpression(ast) : ''
  const displayExpr = ast ? generator.generateDisplay(ast) : ''

  const subExpr: SubExpression = {
    id: uuidv4(),
    name,
    description,
    tokens: clonedTokens,
    compiledExpression: compiledExpr,
    displayExpression: displayExpr,
    createdAt: Date.now(),
    updatedAt: Date.now(),
  }

  subExpressions.value.push(subExpr)
  return { success: true }
}

/** 删除子表达式 */
function removeSubExpression(id: string) {
  const index = subExpressions.value.findIndex(s => s.id === id)
  if (index !== -1) {
    subExpressions.value.splice(index, 1)
  }
}

/** 更新子表达式 */
function updateSubExpression(id: string, updates: Partial<Pick<SubExpression, 'name' | 'description'>>) {
  const subExpr = subExpressions.value.find(s => s.id === id)
  if (subExpr) {
    Object.assign(subExpr, updates, { updatedAt: Date.now() })
  }
}

/**
 * 将子表达式拖入画布时，创建子表达式 Token 或展开
 * @param mode 'collapsed' 显示为一个整体块 | 'expanded' 直接展开为多个 Token
 */
function insertSubExpression(subExprId: string, index: number, mode: 'collapsed' | 'expanded' = 'collapsed') {
  const subExpr = subExpressions.value.find(s => s.id === subExprId)
  if (!subExpr) return

  if (mode === 'collapsed') {
    // 创建一个代表子表达式的单个 Token
    const token = createToken(TokenType.SUB_EXPR, {
      displayText: subExpr.name,
      outputValue: subExpr.compiledExpression,
      subExprId: subExpr.id,
      subExprTokens: JSON.parse(JSON.stringify(subExpr.tokens)),
    })
    insertToken(token, index)
  } else {
    // 展开模式：直接插入子表达式的所有 Token（深拷贝）
    const expandedTokens = subExpr.tokens.map(t => cloneToken(t))
    insertTokens(expandedTokens, index)
  }
}
```

## 12. 键盘快捷键与无障碍支持

### 12.1 快捷键 Composable

```typescript
// ==================== composables/formula/useFormulaKeyboard.ts ====================

import { onMounted, onUnmounted, ref } from 'vue'
import { useFormulaStore } from '@/stores/formulaStore'

export function useFormulaKeyboard() {
  const store = useFormulaStore()

  /** 当前选中的 Token ID */
  const selectedTokenId = ref<string | null>(null)

  /** 当前选中的 Token 索引 */
  const selectedIndex = ref<number>(-1)

  const shortcutMap: Record<string, () => void> = {
    // Ctrl+Z 撤销
    'ctrl+z': () => store.undo(),
    'meta+z': () => store.undo(),

    // Ctrl+Shift+Z / Ctrl+Y 重做
    'ctrl+shift+z': () => store.redo(),
    'meta+shift+z': () => store.redo(),
    'ctrl+y': () => store.redo(),
    'meta+y': () => store.redo(),

    // Delete / Backspace 删除选中
    'delete': () => {
      if (selectedTokenId.value) {
        store.removeToken(selectedTokenId.value)
        selectedTokenId.value = null
      }
    },
    'backspace': () => {
      if (selectedTokenId.value) {
        store.removeToken(selectedTokenId.value)
        selectedTokenId.value = null
      }
    },

    // Ctrl+D 复制选中
    'ctrl+d': () => {
      if (selectedTokenId.value) {
        store.duplicateToken(selectedTokenId.value)
      }
    },
    'meta+d': () => {
      if (selectedTokenId.value) {
        store.duplicateToken(selectedTokenId.value)
      }
    },

    // 方向键导航
    'arrowleft': () => navigateToken(-1),
    'arrowright': () => navigateToken(1),

    // Escape 取消选中
    'escape': () => {
      selectedTokenId.value = null
      selectedIndex.value = -1
    },
  }

  function handleKeyDown(e: KeyboardEvent) {
    // 忽略输入框中的事件
    const target = e.target as HTMLElement
    if (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA') return

    const parts: string[] = []
    if (e.ctrlKey) parts.push('ctrl')
    if (e.metaKey) parts.push('meta')
    if (e.shiftKey) parts.push('shift')
    if (e.altKey) parts.push('alt')
    parts.push(e.key.toLowerCase())

    const combo = parts.join('+')

    // 逐级匹配（meta+shift+z, ctrl+shift+z, delete, etc.）
    for (const [shortcut, handler] of Object.entries(shortcutMap)) {
      if (combo === shortcut || combo.endsWith(shortcut)) {
        e.preventDefault()
        handler()
        return
      }
    }
  }

  function navigateToken(direction: number) {
    if (store.tokens.length === 0) return

    if (selectedIndex.value === -1) {
      // 未选中，选中第一个或最后一个
      selectedIndex.value = direction > 0 ? 0 : store.tokens.length - 1
    } else {
      selectedIndex.value = Math.max(0,
        Math.min(store.tokens.length - 1, selectedIndex.value + direction)
      )
    }

    selectedTokenId.value = store.tokens[selectedIndex.value]?.id || null
  }

  /** 选中指定 Token */
  function selectToken(tokenId: string) {
    selectedTokenId.value = tokenId
    selectedIndex.value = store.tokens.findIndex(t => t.id === tokenId)
  }

  /** 清除选中 */
  function clearSelection() {
    selectedTokenId.value = null
    selectedIndex.value = -1
  }

  onMounted(() => {
    window.addEventListener('keydown', handleKeyDown)
  })

  onUnmounted(() => {
    window.removeEventListener('keydown', handleKeyDown)
  })

  return {
    selectedTokenId,
    selectedIndex,
    selectToken,
    clearSelection,
  }
}
```

```typescript
  version: '1.0'
  tokens: FormulaToken[]
  templateGroups: Array<[string, TemplateGroup]>
  subExpressions: SubExpression[]
  metadata: {
    createdAt: number
    updatedAt: number
    name?: string
    description?: string
  }
}

/**
 * 序列化公式数据
 */
export function serializeFormula(
  tokens: FormulaToken[],
  templateGroups: Map<string, TemplateGroup>,
  subExpressions: SubExpression[],
  metadata?: Partial<SerializedFormula['metadata']>
): SerializedFormula {
  return {
    version: '1.0',
    tokens: JSON.parse(JSON.stringify(tokens)),
    templateGroups: Array.from(templateGroups.entries()),
    subExpressions: JSON.parse(JSON.stringify(subExpressions)),
    metadata: {
      createdAt: Date.now(),
      updatedAt: Date.now(),
      ...metadata,
    },
  }
}

/**
 * 反序列化公式数据
 */
export function deserializeFormula(data: SerializedFormula): {
  tokens: FormulaToken[]
  templateGroups: Map<string, TemplateGroup>
  subExpressions: SubExpression[]
} {
  if (data.version !== '1.0') {
    console.warn(`未知的公式版本: ${data.version}，尝试兼容解析`)
  }

  return {
    tokens: data.tokens || [],
    templateGroups: new Map(data.templateGroups || []),
    subExpressions: data.subExpressions || [],
  }
}

/**
 * 自动保存到 localStorage（防抖）
 */
export function createAutoSaver(key: string, debounceMs: number = 1000) {
  let timer: ReturnType<typeof setTimeout> | null = null

  return {
    save(data: SerializedFormula) {
      if (timer) clearTimeout(timer)
      timer = setTimeout(() => {
        try {
          localStorage.setItem(key, JSON.stringify(data))
        } catch (e) {
          console.error('自动保存失败:', e)
        }
      }, debounceMs)
    },

    load(): SerializedFormula | null {
      try {
        const raw = localStorage.getItem(key)
        return raw ? JSON.parse(raw) : null
      } catch {
        return null
      }
    },

    clear() {
      localStorage.removeItem(key)
    },
  }
}
```

```typescript
// ==================== composables/formula/useFormulaPersistence.ts ====================

import { watch } from 'vue'
import { useFormulaStore } from '@/stores/formulaStore'
import { useFormulaCompiler } from './useFormulaCompiler'
import { serializeFormula, deserializeFormula, createAutoSaver } from '@/utils/formula/serializer'
import { saveFormula, loadFormula, getPredefinedVariables } from '@/api/formulaApi'
import { ElMessage } from 'element-plus'

const AUTO_SAVE_KEY = 'formula_builder_draft'

export function useFormulaPersistence() {
  const store = useFormulaStore()
  const compiler = useFormulaCompiler()
  const autoSaver = createAutoSaver(AUTO_SAVE_KEY, 2000)

  /**
   * 自动保存草稿（watch tokens 变化）
   */
  function enableAutoSave() {
    watch(
      () => store.tokens,
      () => {
        const data = serializeFormula(
          store.tokens,
          store.templateGroups,
          store.subExpressions
        )
        autoSaver.save(data)
      },
      { deep: true }
    )
  }

  /**
   * 从草稿恢复
   */
  function restoreFromDraft(): boolean {
    const data = autoSaver.load()
    if (!data) return false

    const { tokens, templateGroups, subExpressions } = deserializeFormula(data)
    store.tokens = tokens
    store.templateGroups = templateGroups
    store.subExpressions = subExpressions
    store.runValidation()
    return true
  }

  /**
   * 保存到后端
   */
  async function saveToServer(name: string, description?: string) {
    const result = compiler.compile()

    const editorData = serializeFormula(
      store.tokens,
      store.templateGroups,
      store.subExpressions,
      { name, description }
    )

    try {
      const res = await saveFormula({
        name,
        description,
        editorData: JSON.stringify(editorData),
        expression: result.success ? result.expression : '',
        displayExpression: result.success ? result.display : '',
      })

      autoSaver.clear() // 保存成功清除草稿
      ElMessage.success('公式保存成功')
      return res.data.id
    } catch (e: any) {
      ElMessage.error('保存失败: ' + (e.message || '网络错误'))
      throw e
    }
  }

  /**
   * 从后端加载
   */
  async function loadFromServer(id: string) {
    try {
      const res = await loadFormula(id)
      const dto = res.data

      if (dto.editorData) {
        const data = JSON.parse(dto.editorData) as SerializedFormula
        const { tokens, templateGroups, subExpressions } = deserializeFormula(data)
        store.tokens = tokens
        store.templateGroups = templateGroups
        store.subExpressions = subExpressions
        store.runValidation()
      }
    } catch (e: any) {
      ElMessage.error('加载失败: ' + (e.message || '网络错误'))
      throw e
    }
  }

  /**
   * 加载预定义变量
   */
  async function loadPredefinedVariables() {
    try {
      const res = await getPredefinedVariables()
      store.predefinedVariables = res.data
    } catch {
      console.warn('加载预定义变量失败，使用默认配置')
    }
  }

  return {
    enableAutoSave,
    restoreFromDraft,
    saveToServer,
    loadFromServer,
    loadPredefinedVariables,
  }
}
```

```typescript
 * 判断规则：
 * - 如果 SUB 出现在序列开头 → 一元负号
 * - 如果 SUB 前面是运算符 → 一元负号
 * - 如果 SUB 前面是左括号 → 一元负号
 * - 如果 SUB 前面是 IF/THEN/ELSE → 一元负号
 * - 其他情况 → 二元减法运算符
 */
function isUnaryMinus(tokens: FormulaToken[], index: number): boolean {
  if (index === 0) return true

  const prevToken = tokens[index - 1]
  const prevRole = getTokenRole(prevToken.type)

  return [
    TokenRole.BINARY_OP,
    TokenRole.UNARY_PREFIX,
    TokenRole.LPAREN,
    TokenRole.IF,
    TokenRole.THEN,
    TokenRole.ELSE,
  ].includes(prevRole)
}
```

### 14.3 循环引用检测

```typescript
// ==================== engine/formula/circularCheck.ts ====================

import type { SubExpression } from '@/types/formula'
import { TokenType } from '@/types/formula'

/**
 * 检测子表达式是否存在循环引用
 * 
 * @param newSubExprTokens 即将保存的子表达式包含的 Token
 * @param allSubExpressions 现有所有子表达式
 * @param targetSubExprId 如果是更新，传入自身 ID
 * @returns 是否存在循环
 */
export function hasCircularReference(
  newSubExprTokens: FormulaToken[],
  allSubExpressions: SubExpression[],
  targetSubExprId?: string
): boolean {
  // 构建引用图
  const graph = new Map<string, Set<string>>()

  // 已有的子表达式引用关系
  for (const subExpr of allSubExpressions) {
    const refs = new Set<string>()
    for (const token of subExpr.tokens) {
      if (token.type === TokenType.SUB_EXPR && token.subExprId) {
        refs.add(token.subExprId)
      }
    }
    graph.set(subExpr.id, refs)
  }

  // 新表达式的引用
  const newRefs = new Set<string>()
  for (const token of newSubExprTokens) {
    if (token.type === TokenType.SUB_EXPR && token.subExprId) {
      newRefs.add(token.subExprId)
    }
  }

  // 如果是新建：检查新表达式引用的子表达式是否间接引用回新表达式（不可能，因为还没 ID）
  // 如果是更新：检查目标 ID 是否在引用链中出现
  if (!targetSubExprId) return false

  graph.set(targetSubExprId, newRefs)

  // DFS 检测环
  const visited = new Set<string>()
  const recursionStack = new Set<string>()

  function dfs(nodeId: string): boolean {
    visited.add(nodeId)
    recursionStack.add(nodeId)

    const neighbors = graph.get(nodeId) || new Set()
    for (const neighbor of neighbors) {
      if (!visited.has(neighbor)) {
        if (dfs(neighbor)) return true
      } else if (recursionStack.has(neighbor)) {
        return true // 发现环
      }
    }

    recursionStack.delete(nodeId)
    return false
  }

  return dfs(targetSubExprId)
}
```

```typescript
/**
 * 常量值安全校验
 * 防止注入、非法字符
 */
function validateConstantValues(tokens: FormulaToken[]): ValidationError[] {
  const errors: ValidationError[] = []

  const FORBIDDEN_PATTERNS = [
    /[;{}]/,                    // 语句分隔、代码块
    /\b(eval|exec|system)\b/i, // 危险函数名
    /<script/i,                 // XSS
    /\\/,                       // 反斜杠（除了转义场景）
  ]

  for (let i = 0; i < tokens.length; i++) {
    const token = tokens[i]
    if (token.type !== TokenType.CONSTANT) continue

    for (const pattern of FORBIDDEN_PATTERNS) {
      if (pattern.test(token.outputValue)) {
        errors.push({
          tokenId: token.id,
          tokenIndex: i,
          message: '常量值包含非法字符',
          severity: 'error',
        })
        break
      }
    }

    // 数字格式校验（如果看起来像数字但格式不对）
    const val = token.outputValue.trim()
    if (/^\d/.test(val) && isNaN(Number(val))) {
      errors.push({
        tokenId: token.id,
        tokenIndex: i,
        message: '数字格式不正确',
        severity: 'error',
      })
    }
  }

  return errors
}
```

### 16.2 拖拽性能

```typescript
// ==================== composables/formula/useDragPerformance.ts ====================

import { ref } from 'vue'
import { rafThrottle } from '@/utils/throttle'

/**
 * 拖拽性能优化：
 * 1. dragover 节流到 rAF
 * 2. 插入位置计算缓存（鼠标位置没变就不重算）
 * 3. DOM 查询缓存（Token 元素列表在 dragstart 时缓存）
 */
export function useDragPerformance() {
  let cachedTokenRects: DOMRect[] = []
  let lastMouseX = 0
  let lastMouseY = 0
  let lastResult = 0

  /** 在 dragstart 时缓存所有 Token 的位置 */
  function cacheTokenPositions(container: HTMLElement) {
    const elements = container.querySelectorAll<HTMLElement>('[data-token-index]')
    cachedTokenRects = Array.from(elements).map(el => el.getBoundingClientRect())
  }

  /** 高性能插入位置计算 */
  const calculateInsertIndexFast = rafThrottle((mouseX: number, mouseY: number, callback: (index: number) => void) => {
    // 如果鼠标位置没变，直接返回缓存结果
    if (Math.abs(mouseX - lastMouseX) < 2 && Math.abs(mouseY - lastMouseY) < 2) {
      callback(lastResult)
      return
    }

    lastMouseX = mouseX
    lastMouseY = mouseY

    if (cachedTokenRects.length === 0) {
      lastResult = 0
      callback(0)
      return
    }

    // 二分查找最近的位置（Token 水平排列时）
    let closestIndex = 0
    let closestDist = Infinity

    for (let i = 0; i < cachedTokenRects.length; i++) {
      const rect = cachedTokenRects[i]
      const centerX = rect.left + rect.width / 2

      if (mouseX < centerX) {
        const dist = Math.abs(mouseX - rect.left)
        if (dist < closestDist) {
          closestDist = dist
          closestIndex = i
        }
        break // 水平排列，找到第一个中心在右边的就停
      } else {
        const dist = Math.abs(mouseX - rect.right)
        if (dist < closestDist) {
          closestDist = dist
          closestIndex = i + 1
        }
      }
    }

    lastResult = closestIndex
    callback(closestIndex)
  })

  /** 清理缓存 */
  function clearCache() {
    cachedTokenRects = []
    lastMouseX = 0
    lastMouseY = 0
  }

  return {
    cacheTokenPositions,
    calculateInsertIndexFast,
    clearCache,
  }
}
```
